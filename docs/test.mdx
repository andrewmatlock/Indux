    // Test initialization
    const testInit = () => {
      // Initialize theme settings
      const themeSettings = {
        theme: 'light',
        components: {
          header: true,
          sidebar: true
        }
      };

      // Test theme application
      const applyTheme = (theme) => {
        console.log(`Applying theme: ${theme}`);
        return theme === 'light' ? 'light-theme-applied' : 'dark-theme-applied';
      };

      // Test component initialization
      const initComponents = (components) => {
        console.log('Initializing components:', components);
        return Object.keys(components).filter(key => components[key]).length;
      };

      // Run tests
      const output = document.getElementById('test-output');
      output.innerHTML = `
        <p>Theme test: ${applyTheme(themeSettings.theme)}</p>
        <p>Components test: ${initComponents(themeSettings.components)}</p>
      `;
    };

    // Execute test
    testInit();
  </CodeBlock>
